// src/components/TermOfWeekPopup.jsx
import React, { useState } from 'react';
import { Dialog, DialogTitle, DialogContent, DialogActions, Button, Avatar, Typography, Box, IconButton, Snackbar, Alert } from '@mui/material';
import CloseIcon from '@mui/icons-material/Close';
import ThumbUpIcon from '@mui/icons-material/ThumbUp';
import ThumbDownIcon from '@mui/icons-material/ThumbDown';
import Cookies from 'js-cookie'; // npm i js-cookie
import { stringAvatar } from './CommentsSection';
import { initiateNewVote } from '../../../network/TermVoteCalls'; // <-- your POST helper

/**
 * Props
 * -----
 * term          : object | null   // the term data returned by /api/v1/term-of-week
 * open          : boolean        // whether the dialog is visible
 * onClose       : () => void     // called when the user closes the dialog (or after a successful vote)
 *
 * The parent component (InformationHub) decides **when** to render this component
 * and passes the term object that it already fetched.
 */
const TermOfWeekPopup = ({ term, open, onClose }) => {
    const [voteInProgress, setVoteInProgress] = useState(false);
    const [snack, setSnack] = useState({ open: false, severity: 'info', message: '' });

    // -----------------------------------------------------------------
    // 1️⃣  Vote handler – calls the back‑end and then closes the dialog
    // -----------------------------------------------------------------
    const handleVote = async choice => {
        if (!term) return;
        setVoteInProgress(true);
        try {
            // `postVote` should accept termUuid, voteUuid and the choice string
            await initiateNewVote(term.term_uuid, term.vote_uuid, choice);

            setSnack({
                open: true,
                severity: 'success',
                message: 'Your vote was recorded – thanks!'
            });

            // Remember that the user has seen the popup for this week.
            // We store a cookie that expires in 7 days (you can change the ttl).
            Cookies.set('termOfWeekSeen', 'true', { expires: 7 });

            // Let the parent know the popup can be hidden.
            onClose();
        } catch (err) {
            console.error('Vote error', err);
            setSnack({
                open: true,
                severity: 'error',
                message: 'Failed to submit vote.'
            });
        } finally {
            setVoteInProgress(false);
        }
    };

    // -----------------------------------------------------------------
    // 2️⃣  Render – nothing is rendered if there is no term data
    // -----------------------------------------------------------------
    if (!term) return null;

    return (
        <>
            {/* ------------------- Dialog ------------------- */}
            <Dialog open={open} onClose={onClose} maxWidth="sm" fullWidth>
                <DialogTitle
                    sx={{
                        display: 'flex',
                        alignItems: 'center',
                        justifyContent: 'space-between'
                    }}
                >
                    <Box display="flex" alignItems="center">
                        <Avatar {...stringAvatar(term.label)} sx={{ mr: 1 }} />
                        <Typography variant="h6">Term of the Week</Typography>
                    </Box>
                    <IconButton onClick={onClose} size="small">
                        <CloseIcon />
                    </IconButton>
                </DialogTitle>

                <DialogContent dividers>
                    <Typography variant="subtitle1" fontWeight="bold" gutterBottom>
                        {term.label}
                    </Typography>

                    {term.description && (
                        <Typography variant="body2" color="text.secondary" gutterBottom>
                            {term.description}
                        </Typography>
                    )}

                    <Box mt={2} display="flex" alignItems="center" gap={2}>
                        <Typography variant="caption" color="text.secondary">
                            {term.vote_count ?? term.choice_count ?? 0} vote{term.vote_count === 1 ? '' : 's'}
                        </Typography>
                    </Box>
                </DialogContent>

                <DialogActions sx={{ justifyContent: 'center', pb: 2 }}>
                    <Button
                        variant="contained"
                        color="primary"
                        startIcon={<ThumbUpIcon />}
                        disabled={voteInProgress}
                        onClick={() => handleVote('agree')}
                    >
                        Agree
                    </Button>
                    <Button
                        variant="outlined"
                        color="secondary"
                        startIcon={<ThumbDownIcon />}
                        disabled={voteInProgress}
                        onClick={() => handleVote('disagree')}
                    >
                        Disagree
                    </Button>
                </DialogActions>
            </Dialog>

            {/* ------------------- Snackbar ------------------- */}
            <Snackbar
                open={snack.open}
                autoHideDuration={4000}
                onClose={() => setSnack({ ...snack, open: false })}
                anchorOrigin={{ vertical: 'bottom', horizontal: 'center' }}
            >
                <Alert severity={snack.severity} onClose={() => setSnack({ ...snack, open: false })}>
                    {snack.message}
                </Alert>
            </Snackbar>
        </>
    );
};

export default TermOfWeekPopup;
